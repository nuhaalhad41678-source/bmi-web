        <input id="weight" type="number" placeholder="Enter weight in kg">

        <button onclick="calculateBMI()">Calculate</button>

        <div class="result" id="result"></div>
    </div>

    <script>
        async function calculateBMI() {
            const height = document.getElementById("height").value;
            const weight = document.getElementById("weight").value;

            const resultBox = document.getElementById("result");
            resultBox.innerHTML = "Calculating...";

            try {
                const response = await fetch("https://bmi-api-8be0.onrender.com/bmi", {
                    method: "POST",
                    headers: { "Content-Type": "application/json" },
                    body: JSON.stringify({ height: Number(height), weight: Number(weight) })
                });

                const data = await response.json();

                resultBox.innerHTML =
                    "BMI: " + data.bmi + "<br>" +
                    "Status: " + data.status;

            } catch (error) {
                resultBox.innerHTML = "Error connecting to server.";
            }
        }
    </script>
